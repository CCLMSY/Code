#include<iostream>
using std::ostream;
#include<string>
using std::string;
#include<vector>
using std::vector;
#include<cmath>
using std::cin;
using std::cout;
#define endl '\n'
using std::move;

class Point{
public:
    Point(int _x=0,int _y=0):x(_x),y(_y){}
    friend ostream& operator<<(ostream& out,const Point& p);
    Point& operator++(){
        this->x++;
        this->y++;
        return *this;
    }
    Point operator++(int){
        Point p=*this;
        ++(*this);
        return p;
    }
private:
    int x,y;
};

ostream& operator<<(ostream& out,const Point& p){
    out << "(" << p.x << "," << p.y << ")";
    return out;
}

int main(){
    vector<Point> v={{1,2},{3,4},{5,6}};
    cout << v[1]++ << endl;
    for(auto &x:v) cout << x << endl;
    return 0;
}
